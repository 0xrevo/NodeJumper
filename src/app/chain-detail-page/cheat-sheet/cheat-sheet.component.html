Validator

Create New Validator

{{chain.binaryName}} tx staking create-validator \ --amount=1000000{{chain.denomName}} \ --pubkey=$({{chain.binaryName}} tendermint show-validator) \ --moniker=<YOUR_MONIKER> \ --identity=<YOUR_IDENTITY> \ --chain-id={{chain.chainId}} \ --commission-max-change-rate=0.05 \ --commission-max-rate=0.2 \ --commission-rate=0.1 \ --min-self-delegation=1 \ --from={{chain.walletName}} \ --fees={{chain.fees + chain.denomName}} \ --gas=auto \ -y

Edit Existing Validator

{{chain.binaryName}} tx staking edit-validator \ --chain-id={{chain.chainId}} \ --moniker=<YOUR_MONIKER> \ --identity=<YOUR_IDENTITY> \ --commission-rate=0.1 \ --from={{chain.walletName}} \ --fees={{chain.fees + chain.denomName}} \ --gas=auto \ -y

Unjail Validator

Jail Reason

{{chain.binaryName}} query slashing signing-info $({{chain.binaryName}} tendermint show-validator)

List All Active Validators

{{chain.binaryName}} q staking validators -oj --limit=3000 | jq '.validators[] | select(.status=="BOND_STATUS_BONDED")' | jq -r '(.tokens|tonumber/pow(10; 6)|floor|tostring) + " \t " + .description.moniker' | sort -gr | nl

List All Inactive Validators

{{chain.binaryName}} q staking validators -oj --limit=3000 | jq '.validators[] | select(.status=="BOND_STATUS_UNBONDED")' | jq -r '(.tokens|tonumber/pow(10; 6)|floor|tostring) + " \t " + .description.moniker' | sort -gr | nl
